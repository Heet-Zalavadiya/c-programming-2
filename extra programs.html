Pointer and File Handling Programs

SECTION 1: POINTER PROGRAMS

Questions

  1. Calculate average of array using pointers.
  2. Count vowels in string using pointers.
  3. Find minimum element in array using pointers.
  4. Reverse array using pointers.
  5. Perform matrix addition using pointers.

Answers

1. Average of Array

#include <stdio.h>
int main() {
    int arr[5] = {10,20,30,40,50};
    int *p = arr;
    int sum = 0;

    for(int i=0;i<5;i++) {
        sum += *(p+i);
    }

    printf("Average = %.2f", (float)sum/5);
    return 0;
}

2. Count Vowels

#include <stdio.h>
int main() {
    char str[] = "Hello World";
    char *p = str;
    int count = 0;

    while(*p != '\0') {
        if(strchr("aeiouAEIOU", *p)) count++;
        p++;
    }

    printf("Vowels = %d", count);
    return 0;
}

3. Minimum Element

#include <stdio.h>
int main() {
    int arr[5] = {12,5,8,20,3};
    int *p = arr;
    int min = *p;

    for(int i=1;i<5;i++) {
        if(*(p+i) < min)
            min = *(p+i);
    }

    printf("Minimum = %d", min);
    return 0;
}

4. Reverse Array

#include <stdio.h>
int main() {
    int arr[5] = {1,2,3,4,5};
    int *start = arr;
    int *end = arr+4, temp;

    while(start < end) {
        temp = *start;
        *start = *end;
        *end = temp;
        start++;
        end--;
    }

    for(int i=0;i<5;i++)
        printf("%d ", arr[i]);

    return 0;
}

5. Matrix Addition

#include <stdio.h>
int main() {
    int a[2][2]={{1,2},{3,4}}, b[2][2]={{5,6},{7,8}}, c[2][2];
    int *p1=&a[0][0], *p2=&b[0][0], *p3=&c[0][0];

    for(int i=0;i<4;i++)
        *(p3+i)=*(p1+i)+*(p2+i);

    for(int i=0;i<2;i++) {
        for(int j=0;j<2;j++)
            printf("%d ", c[i][j]);
        printf("\n");
    }
    return 0;
}

SECTION 2: FILE HANDLING PROGRAMS

Questions

  1. Write and read data from file.
  2. Count characters in file.
  3. Convert file content to uppercase.
  4. Copy odd lines to another file.
  5. Store and display student records.

Answers

1. Write & Read File

#include <stdio.h>
int main() {
    FILE *fp;
    char text[50];

    fp=fopen("test.txt","w");
    fprintf(fp,"Hello File Handling");
    fclose(fp);

    fp=fopen("test.txt","r");
    fscanf(fp,"%[^\n]",text);
    printf("%s",text);
    fclose(fp);

    return 0;
}

2. Count Characters

#include <stdio.h>
int main() {
    FILE *fp;
    char ch;
    int count=0;

    fp=fopen("test.txt","r");

    while((ch=fgetc(fp))!=EOF)
        count++;

    printf("Characters = %d",count);
    fclose(fp);
    return 0;
}

3. Uppercase File

#include <stdio.h>
#include <ctype.h>

int main() {
    FILE *fp;
    char ch;

    fp=fopen("test.txt","r");

    while((ch=fgetc(fp))!=EOF)
        printf("%c",toupper(ch));

    fclose(fp);
    return 0;
}

4. Copy Odd Lines

#include <stdio.h>
int main() {
    FILE *f1,*f2;
    char line[100];
    int line_no=1;

    f1=fopen("input.txt","r");
    f2=fopen("output.txt","w");

    while(fgets(line,100,f1)) {
        if(line_no%2!=0)
            fputs(line,f2);
        line_no++;
    }

    fclose(f1);
    fclose(f2);
    return 0;
}

5. Student Record

#include <stdio.h>

struct student {
    int roll;
    char name[20];
    float marks;
};

int main() {
    FILE *fp;
    struct student s;

    fp=fopen("student.txt","w");
    scanf("%d %s %f",&s.roll,s.name,&s.marks);
    fprintf(fp,"%d %s %.2f\n",s.roll,s.name,s.marks);
    fclose(fp);

    fp=fopen("student.txt","r");
    while(fscanf(fp,"%d %s %f",&s.roll,s.name,&s.marks)!=EOF)
        printf("%d %s %.2f\n",s.roll,s.name,s.marks);

    fclose(fp);
    return 0;
}
